# Создание директории сборки
mkdir -p build && cd build

# Конфигурация CMake
cmake .. -DCMAKE_BUILD_TYPE=Debug \
         -DBUILD_TESTS=ON \
         -DENABLE_CUDA=OFF \
         -DLLVM_DIR=/usr/lib/llvm-14/cmake

# Сборка
make -j$(nproc)

# Запуск тестов
ctest --output-on-failure

# Установка (опционально)
sudo make install

# Компиляция примеров
cd ..
./build/proxima build examples/hello.prx -o hello.bc
./build/proxima run examples/hello.prx

# Запуск тестов компилятора
./build/tests/unit/lexer_test
./build/tests/unit/parser_test
